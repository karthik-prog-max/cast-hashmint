<body>

  <script src="//www.gstatic.com/cast/sdk/libs/cast_receiver/3/cast_receiver.js"></script>

  <script>

    const context = cast.framework.CastReceiverContext.getInstance();


    let peerConnection;


    // Start the Cast Receiver

    context.start();


    // Handle custom messages for WebRTC signaling

    context.addCustomMessageListener('urn:x-cast:webrtc', (event) => {

      const message = event.data;

      if (message.type === 'offer') {

        // Create PeerConnection and set remote description

        peerConnection = new RTCPeerConnection();

        peerConnection.setRemoteDescription(new RTCSessionDescription(message));


        // Create an answer and send it back to the sender

        peerConnection.createAnswer().then((answer) => {

          peerConnection.setLocalDescription(answer);

          context.sendCustomMessage('urn:x-cast:webrtc', event.senderId, answer);

        });


        // Handle incoming video stream

        peerConnection.ontrack = (event) => {

          const videoElement = document.createElement('video');

          videoElement.srcObject = event.streams[0];

          videoElement.autoplay = true;

          videoElement.style.width = '100%';

          videoElement.style.height = '100%';

          document.body.appendChild(videoElement);

        };

      } else if (message.type === 'ice-candidate') {

        // Add ICE candidate

        peerConnection.addIceCandidate(new RTCIceCandidate(message.candidate));

      }

    });

  </script>

</body>
